What is Gno?

  • Gnolang/GnoVM: comprehensive smart contract suite, based on Golang
  • Gno.land: first L1 using Gnolang
  • built on Tendermint2 in the Cosmos/IBC ecosystem
  • new consensus: proof of contributions
  • aims for fairness, simplicity, security, scalability
  • -> long-term viability

Why Go?

  • a good, established language
  • simple, intuitive
  • type-check safety
  • garbage-collected
  • object-oriented, embedding, composability

Why Gno?

  • optimized for blockchain
  • deterministic execution
  • auto-persisted, without ORM
  • interpreted
  • auto-merkle-ized with rich types
  • built-in rendering
  • auto-generated documentation
  • "std"

render.gno

func greetings(name string) string { // private
    return "Welcome, " + name + "!"
}

func Render(path string) string {
    switch {
    case path == "hello":
        return "world!"
    case strings.HasPrefix(path, "greetings/"):
        name := strings.Split(path, "/")[1] // take string after slash
        name = strings.Title(name)          // capitalize first letter
        return greetings(name)
    }
    return "404"
}

boards/post.gno

// A Post is a "thread" or a "reply" depending on context.
// A thread is a Post of a Board that holds other replies.
type Post struct {
    board       *Board
    id          PostID
    creator     std.Address
    title       string // optional
    body        string
    replies     avl.Tree // Post.id -> *Post
    repliesAll  avl.Tree // Post.id -> *Post (all replies, for top-level posts)
    reposts     avl.Tree // Board.id -> Post.id
    threadID    PostID   // original Post.id
    parentID    PostID   // parent Post.id (if reply or repost)
    repostBoard BoardID  // original Board.id (if repost)
    createdAt   time.Time
    updatedAt   time.Time
}
